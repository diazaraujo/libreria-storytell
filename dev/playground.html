Uso en main.js

window.AtlasReplica = {
  ready: true,
  isStub: false,
  async render(scene, ctx) {
    ctx.hidePlaceholder();
    const { rows } = await AtlasLoad.firstCsv(ctx.config.data_download);
    AtlasLineChart.mount(ctx.chartEl, {
      data: rows.map(d => ({ x: d.year, y: d.rate })),
      x: d => d.x, y: d => d.y,
      yDomain: [0, 100],
      width: ctx.chartEl.clientWidth,
      height: ctx.chartEl.clientHeight,
    });
  }
};
